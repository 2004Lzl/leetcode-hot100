class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:
        ans=0
        def total_list(l1):
            he=0
            for i in l1:
                he+=i
            return he
        for i in range(len(nums)):
            for j in range(i,len(nums)):
                if total_list(nums[i:j+1])==k:
                    ans+=1
        return ans
